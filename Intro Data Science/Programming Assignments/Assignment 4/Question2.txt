# Write a short program in Python which requests a number from the user
# and then tells the user whether or not the number is even.





# is_even is a function that returns true if the number entered mod 2 is 0 and false if not. 
# If a number divided by two produces no remainder, that means it is even. 
def is_even(n):
    if (n % 2 == 0):
        return True
    else: 
        return False
    
    
print('Please enter a number:')

num = int(input())

if (is_even(num)):
    print('{} is even!'.format(num))
else:
    print("{} is odd!".format(num))
