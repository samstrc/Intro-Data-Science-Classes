# Problem 5 The following data set represents randomly sampled temperatures (mea-
# sured in degrees):
# 92, 72, 77, 62, 43, 67, 72, 77, 32, 34, 75, 72, 71, 65, 73, 66, 75, 64, 73, 65, 84, 56, 38
# Write a program in Python using NumPy to find the maximum value, minimum value,
# mean, median, and standard deviation of the data set. In your code, print out a message
# for each of these items, interpreting them in context.
# Also, have the program produce a histogram of the data with a bin width of 10.

import numpy as np
import matplotlib.pyplot as plt

temperatures = [92, 72, 77, 62, 43, 67, 72, 77, 32, 34, 75, 72, 71, 65, 73, 66, 75, 64, 73, 65, 84, 56, 38]
maximum = np.max(temperatures)
minimum = np.min(temperatures)
mean = np.mean(temperatures)
median = np.median(temperatures)
stdDev = np.std(temperatures)

print("Here is our dataset of temperatures in fahrenheit:", temperatures, f"\n\n The hottest recorded temperature is {maximum} and the coldest recorded temperature is {minimum}. The average temperature recorded is {round(mean, 2)}, the median temperature is {round(median, 2)}, and the standard deviation is {round(stdDev, 4)} units from the mean.")

plt.figure()
hist1, edges1 = np.histogram(temperatures)
plt.bar(edges1[:-1], hist1, width=edges1 [1:] -edges1 [:-1])

plt.show()
